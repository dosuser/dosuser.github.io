RentrantLock의 사용상 주의 점

fairness 설정
fairness설정은 Lock이 걸려 대기 중인 쓰레드 중 가장 오래 기다린 쓰레드가 대기중인 다른 쓰레드 보다 먼저 처리 되도록 처리해 주는 것이다.
faireness설정을 안한 이후에 Lock 이후 io작업을 했더니 기아 상태에 걸렸었다.

예제 코드 ( 클래스내 일부 )[#M_더보기 접기

   private transient ReentrantLock lock = new ReentrantLock(true);
    public boolean Lock(String info){
        int r=1;
        try {
            while(lock.tryLock(6000, TimeUnit.MILLISECONDS)==false){
                lockLog.error(Thread.currentThread().getName() + “| “ + info + “ | Can’t aquire Lock | “ + r);
                if(r >= 10){
                    return false;
                }
                ++r;
               
            }
            lockLog.debug(Thread.currentThread().getName() + “ | “ + info +” | Lock” );
            return true;
        } catch (InterruptedException e) {
            lockLog.error(Thread.currentThread().getName() + “| “ + info + “ | Interrupted :” + e.getMessage() +” | “ + r);
            e.printStackTrace();
            return false;
        }
       
    }
    public void Unlock(String info){
        if(lock.isLocked()==true){
           
            lock.unlock();
            lockLog.debug(Thread.currentThread().getName() +” | “ + info + “ | Unlock”);
        }
